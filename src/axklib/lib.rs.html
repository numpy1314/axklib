2//!
3//! This crate exposes a tiny, no_std-compatible trait (`Klib`) that the
4//! platform/board layer must implement. The trait provides a handful of
5//! common kernel helpers such as memory mapping helpers, timing utilities,
6//! and IRQ registration. The implementation is supplied by the platform
7//! (see `modules/axklib-impl`) and consumed by drivers and other modules.
8//!
9//! The crate also provides small convenience modules (`mem`, `time`, `irq`)
10//! that re-export the trait methods with shorter names to make call sites
11//! more ergonomic.
12//!
13//! Example usage:
14//!
15//! ```ignore
16//! // map 4K of device MMIO at physical address `paddr`
17//! let vaddr = axklib::mem::iomap(paddr, 0x1000)?;
18//!
19//! // busy-wait for 100 microseconds
20//! axklib::time::busy_wait(core::time::Duration::from_micros(100));
21//!
22//! // register an IRQ handler
23//! axklib::irq::register(32, my_irq_handler);
24//! ```

38/// The kernel helper trait that platform implementations must provide.
39#[def_extern_trait]
40pub trait Klib {
41    /// Map a physical memory region into the kernel's virtual address space.
42    ///
43    /// Parameters:
44    /// - `addr`: The physical start address of the region to map.
45    /// - `size`: The size in bytes of the region to map. Typically page-aligned.
46    ///
47    /// Returns:
48    /// - `Ok(VirtAddr)` with the virtual address corresponding to the mapped
49    ///   physical region on success.
50    /// - `Err(_)` with an `AxResult` error code on failure.
51    ///
52    /// Notes:
53    /// - The returned `VirtAddr` is page-aligned when the underlying mapping
54    ///   mechanism requires it.
55    /// - The actual mapping behavior is platform-specific; callers should
56    ///   treat this as an allocation-like operation and ensure the mapping
57    ///   is later cleaned up if the platform/ABI requires it.
58    fn mem_iomap(addr: PhysAddr, size: usize) -> AxResult<VirtAddr>;
59
60    /// Busy-wait the current execution context for the provided duration.
61    ///
62    /// This is intended for short delays where sleeping or timer-based
63    /// suspension is not available or not appropriate (for example, very
64    /// early boot or simple spin-waits). Implementations should aim to be
65    /// reasonably accurate for small durations but exact timing guarantees
66    /// are platform-dependent.
67    fn time_busy_wait(dur: Duration);
68
69    /// Enable or disable the edge/level for a platform IRQ.
70    ///
71    /// `irq` is a platform IRQ number. `enabled` selects whether the IRQ
72    /// should be enabled (true) or disabled (false).
73    fn irq_set_enable(irq: usize, enabled: bool);
74
75    /// Register a simple IRQ handler for the given IRQ number.
76    ///
77    /// Returns `true` if the handler was successfully registered, `false`
78    /// otherwise. The exact semantics (e.g. whether multiple handlers are
79    /// allowed) are platform-specific; callers should consult the platform
80    /// implementation.
81    fn irq_register(irq: usize, handler: IrqHandler) -> bool;
82}
